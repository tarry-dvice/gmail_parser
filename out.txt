#include <bits/stdc++.h>

using namespace std;

int main()
{
    short int N, K;
    long int A[N];
    long int s = 0;
    long int k  = 1;
    short int k1 = 0;

    cin >> N >> K;

    for (int i = 0; i < N; i++)
    {
        cin >> A[i];
    }

    sort(A, A + N);

    for (int i = N - 1; i >= 0; i--)
    {
        if (k1 == K)
        {
            k++;
            k1 = 0;
        }

        s += A[i]*k;

        k1++;
    }

    cout << s << "\n";

    return 0;
}
